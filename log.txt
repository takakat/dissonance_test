const pre_evaluation_loop = {
    timeline: [{
        type: jsPsychSurveyLikert,
        css_classes: ['sticky-top-layout'],
        
        preamble: function() {
            const p = jsPsych.evaluateTimelineVariable('path');
            return `<img src="${p}" class="fixed-img">`;
        },
        
        questions: function() {
            const current_id = jsPsych.evaluateTimelineVariable('id');
            let current_questions = [...sd_scale_fixed];

            if (current_id === pre_check_target_id) {
                const insert_index = Math.floor(Math.random() * (current_questions.length + 1));
                current_questions.splice(insert_index, 0, attention_check_item);
            }
            
            return current_questions;
        },
        
        scale_width: 800,
        on_finish: function(data) {
            const res = data.response;
            sd_keys.forEach(key => {
                if(res[key] !== undefined) {
                    data[key] = res[key];
                }
            });
            let sum = (res.beauty||0) + (res.like||0) + (res.good||0) + (res.interest||0);
            
            data.phase = 'pre';
            data.eval_score = sum / 4;
            // 確実に値を取り出すために getAllTimelineVariables を使います
            const current_vars = jsPsych.getAllTimelineVariables();
            data.img_id = current_vars.id;
            data.img_path = current_vars.path;
            
            if (res.attention_check !== undefined) {
                data.passed_check = (res.attention_check === 0);
            }
        }
    }],
    timeline_variables: stimuli_data,
    randomize_order: true
};
timeline.push(pre_evaluation_loop);

// 4. ターゲット選定と事後評価タイムラインの動的生成
// ---------------------------------------------------------
timeline.push({
    type: jsPsychHtmlKeyboardResponse,
    stimulus: 'データ保存中...',
    trial_duration: 500,
    on_finish: function() {
        const all_data = jsPsych.data.get().filter({phase: 'pre'}).values();
        all_data.sort((a, b) => a.eval_score - b.eval_score);
        
        if(all_data.length > 0){
            // ターゲット選定
            TARGET_DATA.id = all_data[0].img_id;
            TARGET_DATA.path = all_data[0].img_path;
            TARGET_DATA.score = all_data[0].eval_score;
            jsPsych.data.addProperties({ target_id: TARGET_DATA.id });

            // 事後評価用リストの作成 (低評価10枚選抜 -> シャッフル)
            const low_eval_trials = all_data.slice(0, 10);
            const post_stimuli = low_eval_trials.map(t => ({
                id: t.img_id,
                path: t.img_path
            }));
            shuffled_post_stimuli = jsPsych.randomization.shuffle(post_stimuli);
            
            // 事後評価用のチェックターゲットを選ぶ
            if (shuffled_post_stimuli.length > 0) {
                const random_stim = jsPsych.randomization.sampleWithoutReplacement(shuffled_post_stimuli, 1)[0];
                post_check_target_id = random_stim.id;
                jsPsych.data.addProperties({ post_check_target_id: post_check_target_id });
                console.log("Post Check ID:", post_check_target_id);
            }
            
            // ★重要: 事後評価のタイムラインを動的に生成してタイムラインに追加
            // const post_evaluation_timeline = create_post_evaluation_timeline(shuffled_post_stimuli);
            // jsPsych.addNodeToEndOfTimeline(post_evaluation_timeline);
        }
    }
});

            // ★ここを修正: デバッグログを追加し、オブジェクトが来た場合の対策を入れる
            preamble: function() {
                // 変数を取得
                let path = jsPsych.timelineVariable('path');
                
                // コンソールで確認（F12キーのConsoleで値が見れます）
                console.log("Phase 3 Image Path:", path);

                // もしpathが文字列ではなくオブジェクトになってしまっていた場合の安全策
                if (typeof path === 'object' && path !== null && path.path) {
                    path = path.path;
                }
                
                return `<img src="${path}" class="fixed-img">`;
            },
            
            questions: function() {
                const current_id = jsPsych.timelineVariable('id');
                let current_questions = [...sd_scale_fixed]; 
                
                if (current_id === post_check_target_id) {
                    const insert_index = Math.floor(Math.random() * (current_questions.length + 1));
                    current_questions.splice(insert_index, 0, attention_check_item);
                }
                return current_questions;
            },
            
            scale_width: 800,
            
            on_finish: function(data) {
                const res = data.response;
                sd_keys.forEach(key => {
                    if(res[key] !== undefined) {
                        data[key] = res[key];
                    }
                });
                let sum = (res.beauty||0) + (res.like||0) + (res.good||0) + (res.interest||0);
                data.phase = 'post'; 
                data.eval_score = sum / 4;
                
                // ★ここも修正: evaluateTimelineVariable ではなく timelineVariable を使う
                data.img_id = jsPsych.timelineVariable('id');
                data.img_path = jsPsych.timelineVariable('path');
                
                data.is_target = (data.img_id === TARGET_DATA.id);
                
                if (res.attention_check !== undefined) {
                    data.passed_check = (res.attention_check === 0);
                }
            }
        }
    ],
    timeline_variables: function() {
        // 生成されたリストの中身を念のため確認
        console.log("Shuffled Stimuli List:", shuffled_post_stimuli);
        return shuffled_post_stimuli;

};
